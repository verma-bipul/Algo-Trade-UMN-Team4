\begin{frame}{Strategy 1: LSTM Portfolio (90\%)}

\Large
A neural network decides how to split money across 4 asset classes \textbf{every day}

\vspace{0.5cm}

\large
\begin{itemize}
    \item \textbf{VTI}:US Stocks
    \item \textbf{SCHZ}:Bonds
    \item \textbf{PDBC}:Commodities
    \item \textbf{VIXM}:Volatility
\end{itemize}

\vspace{0.5cm}

Trained on \textbf{10 years} of daily data (2016--2026)

Optimized for \textbf{Sharpe ratio}:maximizes returns \textit{per unit of risk}

\end{frame}

% ============================================================
\begin{frame}{How the LSTM Works}

\large
\begin{enumerate}
    \item Feeds in last \textbf{50 days} of prices \& returns for all 4 assets
    \item LSTM processes the sequence, learns patterns
    \item Outputs \textbf{4 weights} (e.g., 28\% stocks, 13\% bonds, 30\% commodities, 29\% volatility)
    \item Rebalances portfolio to match
\end{enumerate}

\vspace{0.5cm}

\textbf{Key:} In uncertain markets, the model \textit{automatically} shifts toward bonds, commodities, and volatility

\end{frame}

\begin{frame}{Strategy 2: RSI-2 Mean Reversion on QQQ (5\%)}

\large
\textbf{RSI (Relative Strength Index)} measures how overbought or oversold an asset is

\vspace{0.3cm}

We use a 2-day RSI:computed from the last 2 daily price changes:
$$\text{RSI}(2) = 100 - \frac{100}{1 + \dfrac{\text{average gain over 2 days}}{\text{average loss over 2 days}}}$$

\begin{itemize}
    \item RSI near 0 = extremely oversold (price dropped sharply)
    \item RSI near 100 = extremely overbought (price surged)
\end{itemize}

\vspace{0.3cm}

\textbf{Our rules:}
\begin{itemize}
    \item \textbf{Buy QQQ} when RSI(2) $< 10$:sharp drop, expect a bounce
    \item \textbf{Sell QQQ} when RSI(2) $> 50$:bounce happened, take profit
    \item Typical hold: 1--5 days
\end{itemize}

\end{frame}

\begin{frame}{Strategy 3: Price vs 10-day SMA on SPY (5\%)}

\large
\textbf{SMA (Simple Moving Average)} smooths out daily noise to reveal the trend

\vspace{0.3cm}

The 10-day SMA is the average closing price over the last 10 trading days:
$$\text{SMA}(10) = \frac{\text{Close}_1 + \text{Close}_2 + \cdots + \text{Close}_{10}}{10}$$

\vspace{0.3cm}

\textbf{Our rules:}
\begin{itemize}
    \item \textbf{Buy SPY} when current price $>$ 10-day SMA (uptrend confirmed)
    \item \textbf{Sell SPY} when current price $<$ 10-day SMA (trend broken)
    \item Typical hold: 3--7 days
\end{itemize}

\vspace{0.3cm}

Complements RSI-2: RSI buys \textit{fear} (dips), SMA rides \textit{momentum} (trends)

\end{frame}

\begin{frame}{Risk Management}

\Large
\begin{enumerate}
    \item \textbf{Diversified}:4 asset classes, not just stocks
    \item \textbf{AI-driven defense}:LSTM shifts to safe havens automatically
    \item \textbf{Position sizing}:90/5/5 split limits any single strategy's impact
    \item \textbf{Long-only}:no leverage, no shorts, no options
    \item \textbf{Daily rebalancing}:never holds stale positions
    \item \textbf{All orders before close}:fills at real-time prices
\end{enumerate}
